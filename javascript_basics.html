📦 1. 变量的定义

变量是存储数据的容器。在JavaScript中，我们可以使用 var、let 和 const 来声明变量。

📝 知识点：
  • var - 旧式声明，可重复声明，有作用域问题
  • let - 现代声明，不可重复声明，块级作用域
  • const - 现代声明，用于常量，不可重新赋值
// 变量声明示例 var name = "小明"; // 字符串 let age = 18; // 数字 const PI = 3.14159; // 常量 // 输出结果 console.log("姓名:", name); console.log("年龄:", age); console.log("圆周率:", PI);

🏷️ 2. 变量类型

JavaScript中有多种数据类型，包括基本类型和引用类型。

// 基本数据类型 let name = "张三"; // String - 字符串 let age = 25; // Number - 数字 let isStudent = true; // Boolean - 布尔值 let job = undefined; // Undefined - 未定义 let phone = null; // Null - 空值 // 引用数据类型 let arr = [1, 2, 3]; // Array - 数组 let person = { // Object - 对象 name: "李四", age: 30 }; // 使用 typeof 检查类型 console.log(typeof name); // "string" console.log(typeof age); // "number" console.log(typeof isStudent); // "boolean"

🧮 3. 表达式

表达式是由运算符和操作数组成的计算式，可以产生一个值。

// 算术表达式 let a = 10, b = 3; console.log("加法:", a + b); // 13 console.log("减法:", a - b); // 7 console.log("乘法:", a * b); // 30 console.log("除法:", a / b); // 3.33... console.log("取余:", a % b); // 1 console.log("幂运算:", a ** b); // 1000 // 比较表达式 console.log("10 > 3:", 10 > 3); // true console.log("10 == '10':", 10 == '10'); // true (宽松相等) console.log("10 === '10':", 10 === '10'); // false (严格相等) // 逻辑表达式 console.log("true && false:", true && false); // false console.log("true || false:", true || false); // true console.log("!true:", !true); // false
点击按钮查看输出...

🔀 4. 流程控制

流程控制语句用于决定代码的执行顺序，包括条件语句和循环语句。

📝 条件语句： if...else、switch
📝 循环语句： for、while、do...while
// if...else 条件语句 let score = 85; if (score >= 90) { console.log("优秀"); } else if (score >= 70) { console.log("良好"); } else { console.log("及格"); } // for 循环 console.log("for循环输出1-5:"); for (let i = 1; i <= 5; i++) { console.log(i); } // while 循环 console.log("while循环输出5-1:"); let count = 5; while (count > 0) { console.log(count); count--; }

📞 5. 函数

函数是一段可重复使用的代码块，用于执行特定任务。

// 函数声明 function greet(name) { return "你好，" + name + "！"; } // 函数表达式 const add = function(a, b) { return a + b; }; // 箭头函数 const multiply = (a, b) => a * b; // 使用函数 console.log(greet("小明")); // 你好，小明！ console.log(add(5, 3)); // 8 console.log(multiply(4, 6)); // 24

🎯 综合练习

运用所学的JavaScript知识，完成一个简单的成绩评级系统。

// 综合练习：成绩评级系统 function gradeStudent(score) { if (score < 0 || score > 100) { return "成绩无效"; } let grade = ""; let emoji = ""; if (score >= 90) { grade = "A"; emoji = "🌟"; } else if (score >= 80) { grade = "B"; emoji = "👍"; } else if (score >= 70) { grade = "C"; emoji = "✅"; } else if (score >= 60) { grade = "D"; emoji = "⚠️"; } else { grade = "F"; emoji = "❌"; } return `成绩: ${score}分, 等级: ${grade} ${emoji}`; } // 测试成绩 console.log(gradeStudent(95)); // A级 console.log(gradeStudent(82)); // B级 console.log(gradeStudent(45)); // F级
